--------------------------------------------------
-- Q1. Use DECODE to assign grades (same as CASE)
--------------------------------------------------
SELECT 
    name,
    marks,
    DECODE(
        TRUNC(marks / 10),
        10, 'A',
        9,  'A',
        8,  'B',
        7,  'C',
        'F'
    ) AS grade
FROM students;

--------------------------------------------------
-- Q4. Use DECODE to display full status
--------------------------------------------------
SELECT 
    order_id,
    status,
    DECODE(
        status,
        'P', 'Pending',
        'S', 'Shipped',
        'D', 'Delivered',
        'Unknown'
    ) AS status_description
FROM orders;
